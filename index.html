  <div id="map"></div>


  <script>

    function initMap() {

      const map = new google.maps.Map(document.getElementById("map"), {

        zoom: 10,

        center: { lat: 52.370, lng: 4.895 }, // Bijv. Amsterdam

      });


      // Voorbeeld locaties (vervang door je eigen)

      const locaties = [

        { lat: 52.377, lng: 4.900, titel: "Locatie 1" },

        { lat: 52.360, lng: 4.885, titel: "Locatie 2" },

        { lat: 52.390, lng: 4.920, titel: "Locatie 3" },

      ];


      // Custom icon (vervang door je eigen URL, bijv. een PNG van 32x32 px)

      const customIcon = {

        url: "https://developers.google.com/maps/documentation/javascript/examples/full/images/beachflag.png", // Voorbeeld icon

        scaledSize: new google.maps.Size(40, 40), // Grootte aanpassen

      };


      let i = 0;

      function dropPin() {

        if (i < locaties.length) {

          new google.maps.Marker({

            position: { lat: locaties[i].lat, lng: locaties[i].lng },

            map: map,

            animation: google.maps.Animation.DROP, // Dit is de vallende animatie!

            title: locaties[i].titel,

            icon: customIcon, // Custom pictogram

          });

          i++;

          setTimeout(dropPin, 400); // Vertraging van 400ms tussen pins

        }

      }


      dropPin(); // Start de animatie

    }

  </script>


  <!-- Laad de API met je eigen key -->
